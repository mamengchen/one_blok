**使用lua的时候在vs2017创建运行时堆栈**
```
extern "C"
{
#include <lua.h>
#include<lauxlib.h>
#include<lualib.h>
}


int main(int argc, char* argv[])
{
	lua_State *state = lua_open();
	luaopen_base(state);
	luaL_loadfile(state, "main.lua");
	lua_pcall(state, 0, 0, 0);
	getchar();
	return 0;
}
```
